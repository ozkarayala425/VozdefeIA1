async function send() { const input = document.getElementById("input"); const text = input.value; if (!text) return; addMessage(text, "user"); input.value = ""; const res = await fetch("/api/chat", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ message: text }) }); const data = await res.json(); addMessage(data.reply, "bot"); }
